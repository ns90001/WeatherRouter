34    /**
35     * Test to make sure Ways prints an error when invalid input is passed in.
36     */
37    @Test
38    public void waysErrorTest() {
39      setUp();
40      String out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "0")));
41      assertTrue(out.contains("ERROR:"));
42      out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "0", "0", "100", "100", "50")));
43      assertTrue(out.contains("ERROR:"));
44      out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "0", "0", "100", "WORD")));
45      assertTrue(out.contains("ERROR:"));
46      out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "1", "1", "0", "0")));
47      assertTrue(out.contains("ERROR:"));
48      tearDown();
49    }

65    /**
66     * Simple tests checking ways works properly.
67     */
68    @Test
69    public void simpleWaysTest() {
70      setUp();
71      String out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "41.82", "-71.4", "41.8203", "-71.4")));
72      assertEquals("/w/0\n/w/1\n/w/2\n/w/3", out);
73      out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "41.82", "-71.4", "41.82", "-71.4")));
74      assertEquals("/w/0\n/w/2", out);
75      out = repl.stringRun(new ArrayList<>(Arrays.asList("ways", "0", "-0", "100", "-100")));
76      assertEquals("/w/0\n/w/1\n/w/2\n/w/3\n/w/4\n/w/5\n/w/6", out);
77      tearDown();
78    }
