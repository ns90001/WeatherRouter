57  
58      List<Way> wayList1 = new ArrayList<>();
59      List<Way> wayList2 = new ArrayList<>();
60      List<Way> wayList3 = new ArrayList<>();
61      List<Way> wayList4 = new ArrayList<>();
62  
63      Way way1 = new Way(
64          new MapNode(0, 0, "", mapCmd),
65          new MapNode(0, 0, "", mapCmd),
66          "1",
67          0);
68      Way way2 = new Way(
69          new MapNode(0, 0, "", mapCmd),
70          new MapNode(0, 0, "", mapCmd),
71          "2",
72          0);
73  
74      wayList1.add(way1);
75      wayList1.add(way2);
76      wayList2.add(way1);
77      wayList2.add(way2);
78      wayList3.add(way2);
79      wayList3.add(way1);
80      wayList4.add(way2);
81  
82      // normal test
83      assertEquals(true, routing.sameRoute(wayList1, wayList2));
84  
85      // different order
86      assertEquals(false, routing.sameRoute(wayList1, wayList3));
87  
88      // different number of ways
89      assertEquals(false, routing.sameRoute(wayList3, wayList4));
