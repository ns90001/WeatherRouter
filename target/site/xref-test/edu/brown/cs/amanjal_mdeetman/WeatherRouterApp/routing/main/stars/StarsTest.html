96    @Test
97    public void testNaiveNeighborsTie() {
98      setUp();
99      String out = repl.stringRun(new ArrayList<>(Arrays.asList("naive_neighbors", "3", "\"Sol\"")));
100     assertEquals(out.length(), 5);
101     for (String farStar : new String[]{"1", "5", "6", "7", "8"}) {
102       assertFalse(out.contains(farStar));
103     }
104     out = repl.stringRun(new ArrayList<>(Arrays.asList("naive_neighbors", "1", "0", "2", "0")));
105     assertEquals(out.length(), 1);
106     assertTrue(out.contains("3") || out.contains("5"));
107     tearDown();
108   }
