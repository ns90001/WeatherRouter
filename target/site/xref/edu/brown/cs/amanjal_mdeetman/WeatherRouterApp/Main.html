64    private void run() {
65      // Parse command line arguments
66      OptionParser parser = new OptionParser();

68      parser.accepts("port").withRequiredArg().ofType(Integer.class)
69      .defaultsTo(DEFAULT_PORT);
70      OptionSet options = parser.parse(args);

72      if (options.has("gui")) {
73        runSparkServer((int) options.valueOf("port"));

104   private void runSparkServer(int port) {
105     Spark.port(port);
106     Spark.externalStaticFileLocation("frontend/");
107 
108     Spark.options("/*", (request, response) -> {
109       String accessControlRequestHeaders = request.headers("Access-Control-Request-Headers");
110       if (accessControlRequestHeaders != null) {
111         response.header("Access-Control-Allow-Headers", accessControlRequestHeaders);
112       }
113 
114       String accessControlRequestMethod = request.headers("Access-Control-Request-Method");
115 
116       if (accessControlRequestMethod != null) {
117         response.header("Access-Control-Allow-Methods", accessControlRequestMethod);
118       }
119 
120       return "OK";
121     });

380   private static class ExceptionPrinter implements ExceptionHandler {
381     @Override
382     public void handle(Exception e, Request req, Response res) {
383       res.status(500);
384       StringWriter stacktrace = new StringWriter();
385       try (PrintWriter pw = new PrintWriter(stacktrace)) {
386         pw.println("<pre>");
387         e.printStackTrace(pw);
388         pw.println("</pre>");
389       }
390       res.body(stacktrace.toString());
391     }
