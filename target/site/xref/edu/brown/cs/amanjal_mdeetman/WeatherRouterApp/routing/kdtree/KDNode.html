3   /**
4    * KDNode class which is what will make up the k-d tree.
5    * @param <T> The type of KDPoint which will be contained in the node
6    */
7   public class KDNode<T extends KDPoint> {
8     private final int dimension;
9     private int keyDimension;
10    private double keyCoordinate;
11    private KDNode<T> left;
12    private KDNode<T> right;
13    private final T point;
14  
15    /**
16     * Constructor for KDNode which initializes instance variables.
17     * @param newPoint The 'value' contained in the node
18     * @param depth Depth of this node in the tree.
19     */
20    public KDNode(T newPoint, int depth) {
21      this.point = newPoint;
22      this.dimension = point.getCoords().length;
23      setKeyDimension(depth % dimension);
24      this.left = null;
25      this.right = null;
26    }
27  
28    /**
29     * Sets the left child of this node.
30     * @param leftNode KDNode which will be it's left child
31     * @return returns true if successful and false otherwise
32     */
33    public boolean setLeft(KDNode<T> leftNode) {
34      if (leftNode.getDimension() == dimension) {
35        this.left = leftNode;
36        return true;
37      }
38      return false;
39    }
40  
41    /**
42     * Sets the right child of this node.
43     * @param rightNode KDNode which will be it's right child
44     * @return returns true if successful and false otherwise
45     */
46    public boolean setRight(KDNode<T> rightNode) {
47      if (rightNode.getDimension() == dimension) {
48        this.right = rightNode;
49        return true;
50      }
51      return false;
52    }
53  
54    /**
55     * Gets the left child.
56     * @return KDnode which is the left child
57     */
58    public KDNode<T> getLeft() {
59      return this.left;
60    }
61  
62    /**
63     * Gets the right child.
64     * @return KDNode which is the right child
65     */
66    public KDNode<T> getRight() {
67      return this.right;
68    }
69  
70    /**
71     * Gets the value associated with this node.
72     * @return the value of type T
73     */
74    public T getPoint() {
75      return this.point;
76    }
77  
78    /**
79     * Sets the relevant dimension and coordinate for this point.
80     * @param newKeyDim key dimension for this node.
81     */
82    public void setKeyDimension(int newKeyDim) {
83      this.keyDimension = newKeyDim;
84      setKeyCoordinate(point.getCoords()[newKeyDim]);
85    }
86  
87    /**
88     * returns the relevant dimension for this point.
89     * @return integer representing the index of the relevant coordinate.
90     */
91    public int getKeyDimension() {
92      return this.keyDimension;
93    }
94  
95    /**
96     * Sets the relevant coordinate for this point.
97     * @param keyCoord relevant coordinate.
98     */
99    private void setKeyCoordinate(double keyCoord) {
100     this.keyCoordinate = keyCoord;
101   }
102 
103   /**
104    * Gets the relevant coordinate for this point.
105    * @return relevant coordinate
106    */
107   public double getKeyCoordinate() {
108     return this.keyCoordinate;
109   }
110 
111   /**
112    * Gets the dimension of the node, equivalent to the number of coordinates which
113    * represents its location.
114    * @return dimension
115    */
116   public int getDimension() {
117     return this.dimension;
118   }
119 }
