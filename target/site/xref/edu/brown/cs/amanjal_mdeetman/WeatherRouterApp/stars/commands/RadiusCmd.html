12  /**
13   * Class representing the radius command.
14   */
15  public class RadiusCmd implements ICallable {
16    private final StarsCmd starsCmd;
17  
18    /**
19     * Constructor for RadiusCmd which initializes instance variables.
20     * @param stars instance of the StarsCmd object which contains the list of stars
21     *                 as well as helper methods
22     */
23    public RadiusCmd(StarsCmd stars) {
24      starsCmd = stars;
25    }
26  
27    /**
28     * Executes the radius command.
29     *
30     * @param tokens arguments for the radius command
31     * @return output of the stars within r from the target sorted from closest to
32     * farthest, or an error message if an error occurs
33     */
34    @Override
35    public String run(List<String> tokens) {
36      if (tokens.size() != 2 && tokens.size() != 4) {
37        return "ERROR: radius command should have 2 or 4 arguments";
38      }
39      double[] location = new double[3];
40      double r;
41      try {
42        r = Double.parseDouble(tokens.get(0));
43      } catch (NumberFormatException e) {
44        return "ERROR: first argument should be a number representing the radius";
45      }
46      if (tokens.size() == 2) {
47        String starName = starsCmd.parseStarName(tokens.get(1));
48        if (starName == null) {
49          return "ERROR: "
50            + "second argument should be the star name and should be surrounded in quotation marks";
51        }
52        Star targetStar = starsCmd.findStar(starName);
53        if (targetStar == null) {
54          return "ERROR: star with name " + starName + " not found";
55        } else {
56          return search(r, targetStar.getCoords(), targetStar.getId());
57        }
58      } else {
59        //coordinates instead of star name given
60        for (int i = 0; i < 3; i++) {
61          try {
62            location[i] = Double.parseDouble(tokens.get(i + 1));
63          } catch (NumberFormatException e) {
64            return "ERROR: argument " + i + " should be a number representing a coordinate";
65          }
66        }
67        return search(r, location, -1);
68      }
69    }
70  
71    /**
72     * Performs the radius search with a k-d tree.
73     *
74     * @param r radius around the target where stars should be found
75     * @param location the target location
76     * @param id the id of the star which should be excluded from the output. Should be
77     *           -1 if all stars should be included
78     * @return output of the stars within a set radius from the target
79     */
80    public String search(double r, double[] location, int id) {
81      if (r < 0) {
82        return "ERROR: radius must be non-negative";
83      }
84      KDTree starsTree = starsCmd.getStarsTree();
85      MultiTreeMap<Double, Star> nearStars = starsTree.radiusSearch(r, location, id);
86      if (nearStars.size() == 0) {
87        return "";
88      }
89      StringBuilder out = new StringBuilder();
90      while (true) {
91        Map.Entry<Double, List<Star>> minEntry = nearStars.pollFirstEntry();
92        if (minEntry == null) {
93          return out.substring(0, out.length() - 1);
94        }
95        for (KDPoint star : minEntry.getValue()) {
96          out.append(star.getId()).append("\n");
97        }
98      }
99    }
100 }
